44void MyPrimaryGenerator::GeneratePrimaries(G4Event *anEvent)
45{
46 G4ParticleDefinition* particle = nullptr;
47 G4double energy;
48
49 switch(fModeType)
50 {
51 case 0: // Standard mode
52 default:
53 if(fModeType != 0) G4cerr << "Not valid mode! Standard mode is selected!" << G4endl;
54 // Set gamma as primary particle
55 particle = G4ParticleTable::GetParticleTable()->FindParticle("gamma");
56 fParticleGun->SetParticleDefinition(particle);
57
58 // Sample the energy from Gaus
59 energy = G4RandGauss::shoot(fMeanEnergy, fSigmaEnergy);
60 fParticleGun->SetParticleEnergy(energy);
61
62 switch(fBeamType)
63 {
64 case 0: // Pointlike beam
65 default:
66 if(fBeamType != 0) G4cerr << "Not valid beam type! Pointlike beam is selected!" << G4endl;
67 break;
68
69 case 1: // Spread beam
70 PrimariesForSpreadBeam();
71 break;
72
73 case 2: // Circle beam
74 PrimariesForCircleBeam();
75 break;
76 }
77 break;
78
79 case 1: // LED-System mode
80 PrimariesForLEDMode();
81 break;
82
83 case 2: // Lu decay mode
84 PrimariesForLuDecayMode();
85 break;
86
87 }
88
89 // Finally generate the primary vertex
90 fParticleGun->GeneratePrimaryVertex(anEvent);
91}

95void MyPrimaryGenerator::PrimariesForSpreadBeam()
96{
97 // Set position at the center
98 G4ThreeVector pos(0., 0., 0.);
99
100 // Sample the momentum direction uniformly within the given range
101 G4double fThetaMax = std::atan(fRadiusSpread/GS::zFrontFaceScintillator);
102 G4double fCosThetaMax = std::cos(fThetaMax);
103
104 G4double cosTheta = 1-G4UniformRand()*(1-fCosThetaMax);
105 G4double sinTheta = std::sqrt(1. - cosTheta*cosTheta);
106 G4double phi = CLHEP::twopi*G4UniformRand();
107
108 G4double ux = sinTheta*std::cos(phi);
109 G4double uy = sinTheta*std::sin(phi);
110 G4double uz = cosTheta;
111
112 G4ThreeVector mom(ux, uy, uz);
113
114 // Set fParticleGun
115 fParticleGun->SetParticlePosition(pos);
116 fParticleGun->SetParticleMomentumDirection(mom);
117}

134void MyPrimaryGenerator::PrimariesForLEDMode()
135{
136 G4ThreeVector posLed;
137 G4ThreeVector momLed;
138
139 // Which face
140 G4double posZLed;
141 if(fChooseFrontorBack=="F") posZLed = GS::zFrontFaceScintillator-GS::halfheightLightGuide;
142 else if(fChooseFrontorBack=="B") posZLed = GS::zBackFaceScintillator+GS::halfheightLightGuide;
143 else
144 {
145 G4cerr << "Option for detector face not valid. 'Front' has been setted" << G4endl;
146 posZLed = GS::zFrontFaceScintillator-GS::halfheightLightGuide;
147 }
148
149 // Which LED
150 if(fSwitchOnLED=="u") posLed = G4ThreeVector(0, GS::radiusLightGuide-GS::depthLED, posZLed);
151 else if(fSwitchOnLED=="d") posLed = G4ThreeVector(0, -GS::radiusLightGuide+GS::depthLED, posZLed);
152 else if(fSwitchOnLED=="l") posLed = G4ThreeVector(GS::radiusLightGuide-GS::depthLED, 0, posZLed);
153 else if(fSwitchOnLED=="r") posLed = G4ThreeVector(-GS::radiusLightGuide+GS::depthLED, 0, posZLed);
154 else
155 {
156 G4cerr << "Option for LED not valid. 'up' has been setted" << G4endl;
157 posLed = G4ThreeVector(0, GS::radiusLightGuide-GS::depthLED, posZLed);
158 }
159
160 // Isotropic emission (not accurate, should be Lambert's cosine law,
161 // but need G4GeneralParticleSource for this)
162 G4double cosTheta = 2*G4UniformRand() - 1.;
163 G4double phi = CLHEP::twopi*G4UniformRand();
164 G4double sinTheta = std::sqrt(1. - cosTheta*cosTheta);
165 G4double ux = sinTheta*std::cos(phi);
166 G4double uy = sinTheta*std::sin(phi);
167 G4double uz = cosTheta;
168 momLed = G4ThreeVector(ux,uy,uz);
169
170 // Set everything in fParticleGun
171 G4ParticleDefinition* particle = G4ParticleTable::GetParticleTable()->FindParticle("opticalphoton");
172 fParticleGun->SetParticleDefinition(particle);
173 fParticleGun->SetParticlePosition(posLed);
174 fParticleGun->SetParticleMomentumDirection(momLed); fParticleGun->SetParticleEnergy(GS::energyLED);
175}

179void MyPrimaryGenerator::PrimariesForLuDecayMode()
180{
181 G4ThreeVector posDecay;
182 G4ThreeVector momDecay;
183
184 // Compute random position inside crystal
185 G4double posZ = GS::zScintillator + 2*(G4UniformRand()-0.5)*GS::halfheightScintillator;
186 G4double posR = GS::radiusScintillator*std::sqrt(G4UniformRand());
187 G4double posPhi = CLHEP::twopi*G4UniformRand();
188
189 posDecay = G4ThreeVector(posR*std::cos(posPhi), posR*std::sin(posPhi), posZ);
190
191 // Isotropic emission
192 G4double cosTheta = 2*G4UniformRand() - 1.;
193 G4double phi = CLHEP::twopi*G4UniformRand();
194 G4double sinTheta = std::sqrt(1. - cosTheta*cosTheta);
195 G4double ux = sinTheta*std::cos(phi);
196 G4double uy = sinTheta*std::sin(phi);
197 G4double uz = cosTheta;
198
199 momDecay = G4ThreeVector(ux,uy,uz);
200
201 // Set everything in fParticleGun
202 G4int Z = 71, A = 176;
203 G4ParticleDefinition* ion = G4IonTable::GetIonTable()->GetIon(Z, A, 0.*keV);
204 fParticleGun->SetParticleDefinition(ion);
205 fParticleGun->SetParticleCharge(0.*eplus);
206 fParticleGun->SetParticlePosition(posDecay);
207 fParticleGun->SetParticleMomentumDirection(momDecay); fParticleGun->SetParticleEnergy(1*eV);
208}
