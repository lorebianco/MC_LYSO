35void MyPrimaryGenerator::GeneratePrimaries(G4Event *anEvent)
36{
37 G4ThreeVector pos;
38 G4ThreeVector mom;
39
40 if(!fIsCalibrationEnabled) // Standard mode
41 {
42 pos = G4ThreeVector(0., 0., 0.);
43
44 if(!fIsSpreadEnabled) // No spread
45 {
46 mom = G4ThreeVector(0., 0., 1.);
47 }
48
49 if(fIsSpreadEnabled) // With spread
50 {
51 // Sample the momentum direction uniformly within the given range
52 G4double fThetaMax = std::atan(fRadiusSpread/GS::zFrontFaceScintillator);
53 G4double fCosThetaMax = std::cos(fThetaMax);
54
55 G4double cosTheta = 1-G4UniformRand()*(1-fCosThetaMax);
56 G4double sinTheta = std::sqrt(1. - cosTheta*cosTheta);
57 G4double phi = CLHEP::twopi*G4UniformRand();
58
59 G4double ux = sinTheta*std::cos(phi);
60 G4double uy = sinTheta*std::sin(phi);
61 G4double uz = cosTheta;
62
63 mom = G4ThreeVector(ux, uy, uz);
64 }
65
66 // Sample the energy from Gaus
67 G4double energy = G4RandGauss::shoot(fMeanEnergy, fSigmaEnergy);
68
69 // Set everything in fParticleGun
70 G4ParticleDefinition* particle = G4ParticleTable::GetParticleTable()->FindParticle("gamma");
71 fParticleGun->SetParticleDefinition(particle);
72 fParticleGun->SetParticlePosition(pos);
73 fParticleGun->SetParticleMomentumDirection(mom);
74 fParticleGun->SetParticleEnergy(energy);
75 }
76
77 if(fIsCalibrationEnabled) // Calibration mode
78 {
79 G4double posZLed;
80
81 // Which face
82 if(fChooseFrontorBack=="F") posZLed = GS::zFrontFaceScintillator-GS::halfheightLightGuide;
83 else if(fChooseFrontorBack=="B") posZLed = GS::zBackFaceScintillator+GS::halfheightLightGuide;
84 else
85 {
86 G4cerr << "Option for detector face not valid. 'Front' has been setted" << G4endl;
87 posZLed = GS::zFrontFaceScintillator-GS::halfheightLightGuide;
88 }
89
90 // Which LED
91 if(fSwitchOnLED=="u") pos = G4ThreeVector(0, GS::radiusLightGuide-GS::depthLED, posZLed);
92 else if(fSwitchOnLED=="d") pos = G4ThreeVector(0, -GS::radiusLightGuide+GS::depthLED, posZLed);
93 else if(fSwitchOnLED=="l") pos = G4ThreeVector(GS::radiusLightGuide-GS::depthLED, 0, posZLed);
94 else if(fSwitchOnLED=="r") pos = G4ThreeVector(-GS::radiusLightGuide+GS::depthLED, 0, posZLed);
95 else
96 {
97 G4cerr << "Option for LED not valid. 'up' has been setted" << G4endl;
98 pos = G4ThreeVector(0, GS::radiusLightGuide-GS::depthLED, posZLed);
99 }
100
101 // Isotropic emission (not accurate, should be Lambert's cosine law,
102 // but need G4GeneralParticleSource upgrade for this)
103 G4double cosTheta = 2*G4UniformRand() - 1.;
104 G4double phi = CLHEP::twopi*G4UniformRand();
105 G4double sinTheta = std::sqrt(1. - cosTheta*cosTheta);
106 G4double ux = sinTheta*std::cos(phi);
107 G4double uy = sinTheta*std::sin(phi);
108 G4double uz = cosTheta;
109 mom = G4ThreeVector(ux,uy,uz);
110
111 // Set everything in fParticleGun
112 G4ParticleDefinition* particle = G4ParticleTable::GetParticleTable()->FindParticle("opticalphoton");
113 fParticleGun->SetParticleDefinition(particle);
114 fParticleGun->SetParticlePosition(pos);
115 fParticleGun->SetParticleMomentumDirection(mom); fParticleGun->SetParticleEnergy(GS::energyLED);
116 }
117
118 // Finally generate the primary vertex
119 fParticleGun->GeneratePrimaryVertex(anEvent);
120}
