134 G4VPhysicalVolume *MyDetectorConstruction::Construct()
135 {
136  // Construct the World
137  solidWorld = new G4Box("solidWorld", GS::halfXsideWorld, GS::halfYsideWorld, GS::halfZsideWorld);
138  logicWorld = new G4LogicalVolume(solidWorld, fAir, "logicWorld");
139  physWorld = new G4PVPlacement(0, G4ThreeVector(0., 0., 0.), logicWorld, "physWorld", 0, false, 0, true);
140 
141  // Construct the scintillator crystal
142  solidScintillator = new G4Tubs("solidScintillator", 0.*cm, GS::radiusScintillator, GS::halfheightScintillator, 0.*deg, 360.*deg);
143  logicScintillator = new G4LogicalVolume(solidScintillator, fLYSO, "logicScintillator");
144  physScintillator = new G4PVPlacement(0, G4ThreeVector(GS::xScintillator, GS::yScintillator, GS::zScintillator), logicScintillator, "physScintillator", logicWorld, false, 0, true);
145 
146  // Assign the logic scoring volume to the crystal
147  fScoringVolume = logicScintillator;
148 
149  // Construct a coating for the crystal and the lightguide (if present)
150  solidCoating = new G4Tubs("solidCoating", (GS::radiusScintillator + GS::coating_space), (GS::radiusScintillator + GS::coating_space + GS::coating_thickness), GS::halfheightScintillator+2*(GS::halfheightLightGuide*fIsLightGuide), 0.*deg, 360.*deg);
151  logicCoating = new G4LogicalVolume(solidCoating, fAluminium, "logicCoating");
152  physCoating = new G4PVPlacement(0, G4ThreeVector(GS::xScintillator, GS::yScintillator, GS::zScintillator), logicCoating, "physCoating", logicWorld, false, 0, true);
153 
154  // Construct lightguides, PCBs and endcaps -if setted-
155  if(fIsLightGuide) ConstructLightGuide();
156  if(fIsPCB) ConstructPCB();
157  if(fIsEndcap) ConstructEndcap();
158 
159  // Construct the package of the SIPM
160  solidPackageSiPM = new G4Box("solidPackageSiPM", GS::halfXsidePackageSiPM, GS::halfYsidePackageSiPM, GS::halfZsidePackageSiPM);
161  logicPackageSiPM = new G4LogicalVolume(solidPackageSiPM, fFR4, "logicPackageSiPM");
162 
163  // Construct the window of the SIPM and put it inside the package
164  solidWindowSiPM = new G4Box("solidWindowSiPM", GS::halfXsideWindowSiPM, GS::halfYsideWindowSiPM, GS::halfZsideWindowSiPM);
165  logicWindowSiPM = new G4LogicalVolume(solidWindowSiPM, fEpoxy, "logicWindowSiPM");
166  physWindowSiPM = new G4PVPlacement(0, G4ThreeVector(GS::xWindowSiPM, GS::yWindowSiPM, GS::zWindowSiPM), logicWindowSiPM, "physWindowSiPM", logicPackageSiPM, false, 0, true);
167 
168  // Construct the silicon layer and put it inside the window
169  solidDetector = new G4Box("solidDetector", GS::halfXsideDetector, GS::halfYsideDetector, GS::halfZsideDetector);
170  logicDetector = new G4LogicalVolume(solidDetector, fSilicon, "logicDetector");
171  physDetector = new G4PVPlacement(0, G4ThreeVector(GS::xDetector, GS::yDetector, GS::zDetector), logicDetector, "physDetector", logicWindowSiPM, false, 0, true);
172 
173  // Define the arrangement of SiPMs
174  G4int indexDetector = 0;
175  for(G4int i = 0; i<13; i++)
176  {
177  if(i==0 || i==12)
178  {
179  for(G4int j = 0; j<3; j++)
180  {
181  PositionSiPMs(i, j, 1, indexDetector);
182  indexDetector++;
183  }
184  }
185  if(i==1 || i==11)
186  {
187  for(G4int j = 0; j<7; j++)
188  {
189  PositionSiPMs(i, j, 3, indexDetector);
190  indexDetector++;
191  }
192  }
193  if(i==2 || i==10)
194  {
195  for(G4int j = 0; j<9; j++)
196  {
197  PositionSiPMs(i, j, 4, indexDetector);
198  indexDetector++;
199  }
200  }
201  if(i>=3 && i<=9)
202  {
203  for(G4int j = 0; j<11; j++)
204  {
205  PositionSiPMs(i, j, 5, indexDetector);
206  indexDetector++;
207  }
208  }
209  }
210 
211  // Add visualization attributes
212  DefineVisAttributes();
213 
214  // Always return physWorld
215  return physWorld;

230 void MyDetectorConstruction::ConstructLightGuide()
231 {
232  G4Material *fLightGuideMaterial = 0;
233 
234  // Solid light guide as cilinder
235  G4Tubs *solidLightGuide = new G4Tubs("solidLightGuide", 0, GS::radiusLightGuide, GS::halfheightLightGuide, 0.*deg, 360.*deg);
236 
237  // Drill 4 holes in the light guide for LEDs
238  G4Tubs *solidHoleUP = new G4Tubs("solidHoleLightGuide", 0, GS::radiusHole, GS::depthHole, 0.*deg, 360*deg);
239  G4Tubs *solidHoleDOWN = new G4Tubs("solidHoleLightGuide", 0, GS::radiusHole, GS::depthHole, 0.*deg, 360*deg);
240  G4Tubs *solidHoleRIGHT = new G4Tubs("solidHoleLightGuide", 0, GS::radiusHole, GS::depthHole, 0.*deg, 360*deg);
241  G4Tubs *solidHoleLEFT = new G4Tubs("solidHoleLightGuide", 0, GS::radiusHole, GS::depthHole, 0.*deg, 360*deg);
242 
243  G4Rotate3D rotXholeUP(90*deg, G4ThreeVector(1, 0, 0));
244  G4Translate3D transYholeUP(G4ThreeVector(0, GS::radiusLightGuide, 0));
245  G4Transform3D transformHoleUP = (transYholeUP)*(rotXholeUP);
246 
247  G4Rotate3D rotXholeDOWN(90*deg, G4ThreeVector(1, 0, 0));
248  G4Translate3D transYholeDOWN(G4ThreeVector(0, -GS::radiusLightGuide, 0));
249  G4Transform3D transformHoleDOWN = (transYholeDOWN)*(rotXholeDOWN);
250 
251  G4Rotate3D rotYholeRIGHT(90*deg, G4ThreeVector(0, 1, 0));
252  G4Translate3D transYholeRIGHT(G4ThreeVector(GS::radiusLightGuide, 0, 0));
253  G4Transform3D transformHoleRIGHT = (transYholeRIGHT)*(rotYholeRIGHT);
254 
255  G4Rotate3D rotYholeLEFT(90*deg, G4ThreeVector(0, 1, 0));
256  G4Translate3D transYholeLEFT(G4ThreeVector(-GS::radiusLightGuide, 0, 0));
257  G4Transform3D transformHoleLEFT = (transYholeLEFT)*(rotYholeLEFT);
258 
259  G4MultiUnion *solidHoles = new G4MultiUnion("solidHoles");
260  solidHoles->AddNode(*solidHoleUP, transformHoleUP);
261  solidHoles->AddNode(*solidHoleDOWN, transformHoleDOWN);
262  solidHoles->AddNode(*solidHoleRIGHT, transformHoleRIGHT);
263  solidHoles->AddNode(*solidHoleLEFT, transformHoleLEFT);
264  solidHoles->Voxelize();
265 
266  solidDrilledLightGuide = new G4SubtractionSolid("solidDrilledLightGuide", solidLightGuide, solidHoles);
267 
268  // Options for material
269  switch(nLightGuideMat)
270  {
271  case 1: // Plexiglass
272  fLightGuideMaterial = fPlexiglass;
273  break;
274  case 2: // Sapphire
275  fLightGuideMaterial = fSapphire;
276  break;
277  default:
278  G4cerr << "Option for lightguide's material not valid. \"Plexiglass\" has been setted" << G4endl;
279  fLightGuideMaterial = fPlexiglass;
280  }
281 
282  // Logic and phys volumes for light guides
283  logicLightGuide = new G4LogicalVolume(solidDrilledLightGuide, fLightGuideMaterial, "logicLightGuide");
284  physFrontLightGuide = new G4PVPlacement(0, G4ThreeVector(0., 0., GS::zFrontFaceScintillator-GS::halfheightLightGuide), logicLightGuide, "physFrontLightGuide", logicWorld, false, 0, true);
285  physBackLightGuide = new G4PVPlacement(0, G4ThreeVector(0., 0., GS::zBackFaceScintillator+GS::halfheightLightGuide), logicLightGuide, "physBackLightGuide", logicWorld, false, 1, true);
286 }
